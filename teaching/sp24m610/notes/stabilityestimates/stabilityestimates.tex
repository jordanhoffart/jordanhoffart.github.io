Consider the following boundary value problem.
\begin{problem}
  Find a function $u : [0,1] \to \mathbb R$ such that
  \begin{align}
    \label{eq:ode} -u'' & = f, \\
    \label{eq:bc} u(0) - u'(0) & = \gamma, \\
    u(1) & = 0.
  \end{align}
\end{problem}
Here, $f \in L^2(0,1)$ and $\gamma \in \mathbb R$ are the \emph{data} of the problem that have been given to us.
This problem is well-defined for $u \in H^2(0,1)$.

The associated weak formulation follows from multiplying \eqref{eq:ode} by a smooth function $v : [0,1] \to \mathbb R$ with $v(1) = 0$, integrating by parts, and applying the boundary condition \eqref{eq:bc}.
\begin{problem}[Weak formulation]\label{prob:weak}
  Find $u \in H_r^1(0,1)$ such that
  \begin{equation}
    a(u,v) = F(v)
  \end{equation}
  for all $v \in H_r^1(0,1)$, where
  \begin{align}
    a(u,v) & = \int_0^1u'v', \\
    F(v) &= \int_0^1fv + \gamma v(0), \\
    H^1_r(0,1) & = \{ v \in H^1(0,1) : v(1) = 0 \}.
  \end{align}
\end{problem}

We have the following Poincar\'e inequality on $H^1_r(0,1)$.
\begin{theorem}[Poincar\'e Inequality]
  There exists $C_P > 0$ such that
  \begin{equation}
    \|u\|_{L^2(0,1)} \leq C_P \|u'\|_{L^2(0,1)}
  \end{equation}
  for all $u \in H^1_r(0,1)$.
\end{theorem}
Therefore, the $H^1$ seminorm \[u \mapsto |u|_{H^1(0,1)} := \|u'\|_{L^2(0,1)} \] is a norm on $H^1_r(0,1)$ that is equivalent to the usual $H^1$ norm.
We endow $H^1_r(0,1)$ with this norm.
Then the bilinear form $a$ is coercive on $H^1_r(0,1)$ with respect to this norm:
\begin{equation}\label{eq:coercive}
  a(u,u) = |u|_{H^1(0,1)}^2 + |u(0)|^2 \geq |u|_{H^1(0,1)}^2.
\end{equation}

We also have the following Sobolev embedding that holds on $H^1(0,1)$.
\begin{theorem}[Sobolev Embedding]
  The inclusion $H^1(0,1) \subset C([0,1])$ holds, and there exists $C > 0$ such that
  \begin{equation}
    \sup_{x\in[0,1]}|u(x)| \leq C\|u\|_{H^1(0,1)}
  \end{equation}
  for all $u \in H^1(0,1)$.
\end{theorem}
In particular, due to the equivalence of norms, this also holds for all $u \in H^1_r(0,1)$ with the $H^1$ seminorm (and a different constant that we will still denote by $C$).

Therefore, we have that there exists $C > 0$ such that
\begin{equation}
  |u(0)| \leq C |u|_{H^1(0,1)}
\end{equation}
for all $u \in H^1_r(0,1)$.
Using this, we have that $F$ is continuous on $H^1_r(0,1)$ with respect to this norm:
\begin{align}\label{eq:continuous}
  |F(u)| & \leq \|f\|_{L^2(0,1)}\|u\|_{L^2(0,1)} + |\gamma| |u(0)| \\ & \leq \left(C_P\|f\|_{L^2(0,1)} + C|\gamma|\right)|u|_{H^1(0,1)}
\end{align}

Combining our observations gives us the following stability estimate.
\begin{theorem}[Energy estimate]
  If $u \in H^1_r(0,1)$ solves Problem \ref{prob:weak}, then
  \begin{equation}
    |u|_{H^1(0,1)} \leq C_P\|f\|_{L^2(0,1)} + C |\gamma|.
  \end{equation}
\end{theorem}
\begin{proof}
  \begin{equation}
    |u|_{H^1(0,1)}^2 \leq a(u,u) = F(u) \leq \left(C_P\|f\|_{L^2(0,1)} + C|\gamma|\right)|u|_{H^1(0,1)}.
  \end{equation}
\end{proof}
Now the question we want to investigate is the following: why is this called a \emph{stability} estimate?
The idea of stability in a problem is that small changes in the data should yield only small changes in the solution.
Here, the data of the problem is $f$ and $\gamma$, so by small changes in the data, we mean that we now consider the same problem but with $f$ and $\gamma$ replaced by $\widetilde f \in L^2(0,1)$ and $\widetilde \gamma \in \mathbb R$ such that
\begin{align}
  \|f - \widetilde f\|_{L^2(0,1)} & < \varepsilon, \\
  |\gamma - \widetilde \gamma| & < \varepsilon,
\end{align}
where $\varepsilon > 0$ is small.
Now we suppose that we have a solution $\widetilde u \in H^1_r(0,1)$ to Problem \ref{prob:weak} with $\widetilde f$ and $\widetilde \gamma$ as the given data: for all $v \in H^1_r(0,1)$,
\begin{equation}
  a(\widetilde u,v) = \widetilde F(v),
\end{equation}
where
\begin{align}
  \widetilde F(v) & = (\widetilde f, v)_{L^2(0,1)} + \widetilde \gamma v(0).
\end{align}

We now want to estimate $|u - \widetilde u|_{H^1(0,1)}$.
To do that, we have that
\begin{align}
  a(u,v) - a(\widetilde u,v) & = F(v) - \widetilde F(v) \\
                             & = (f - \widetilde f, v)_{L^2(0,1)} + \left(|\gamma| - |\widetilde \gamma|\right)v(0) \\
                             & \leq \left(C_P\|f - \widetilde f\|_{L^2(0,1)} + C\left(|\gamma-\widetilde\gamma|\right)\right)|v|_{H^1(0,1)} \\
                             & \leq (C_P + C)\varepsilon |v|_{H^1(0,1)}
\end{align}
for all $v \in H^1_r(0,1)$.
On the other hand, we have that
\begin{align}
  a(u,v) - a(\widetilde u, v) & = a(u - \widetilde u, v)
\end{align}
for all $v \in H^1_r(0,1)$.
Setting $v = u - \widetilde u$ and recalling that $a$ is coercive on $H^1_r(0,1)$ gives us
\begin{align}
  a(u-\widetilde u, u - \widetilde u) \geq |u - \widetilde u|^2_{H^1(0,1)}
\end{align}

Combining these gives us the following theorem.
\begin{theorem}[Stability]
  If $u \in H^1_r(0,1)$ solves Problem \ref{prob:weak} with data $f$ and $\gamma$, if $\widetilde u \in H^1_r(0,1)$ solves Problem \ref{prob:weak} with data $\widetilde f$ and $\widetilde \gamma$, and if the errors in the data are such that
  \begin{align}
    \|f-\widetilde f\|_{L^2(0,1)} < \varepsilon, \\
    |\gamma - \widetilde \gamma| < \varepsilon,
  \end{align}
  then the error in the solutions is
  \begin{equation}
    |u-\widetilde u|_{H^1(0,1)} \leq (C_P + C)\varepsilon.
  \end{equation}
\end{theorem}
